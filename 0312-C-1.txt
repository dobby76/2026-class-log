#include <stdio.h>

int main()
{
    int n;
    
    while (1)
    
   {
    printf("숫자를 입력하세요: ");
    scanf("%d", &n);
    
    if(n == 10)
    {
        printf("정답입니다\n");
        break;
    }
    
    else if(n > 10)
    {
        printf("너무 높아요\n");
    }
    
    else
    {
        printf("너무 낮아요\n");
    }
   }
    return 0;
}
