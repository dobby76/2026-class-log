#include <stdio.h>

int main()
{
    int fd;
    
    printf("원하시는 음식 번호를 입력하세요(1:치킨 2:피자 3:떡볶이 4:햄버거 5~7:랜덤): ");
    scanf("%d", &fd);
    
    if(fd == 1)
    {
        printf("치킨을 선택했습니다.");
    }
    
    else if(fd == 2)
    {
        printf("피자를 선택했습니다.");
    }
    
    else if(fd == 3)
    {
        printf("떡볶이를 선택했습니다.");
    }
    
    else if(fd == 4)
    {
        printf("햄버거를 선택했습니다.");
    }
    
    else if(5<= fd && 7>= fd) //5이상 7 이하를 만족하는지 확인 할 수 있는 코드 파이썬과 달리 &&을 붙여야한다.
    {
        printf("랜덤");
    }
    
    else
    {
        printf("입력 오류");
    }

    return 0;
}
