int main()
{
    char blood;
    
    printf("혈액형 입력(A,B,O): ");
    scanf("%c", &blood);
    
    switch(blood)
    {
        case 'A':
        case 'a':
        printf("A형입니다.\n");
        break;
        
        case 'B':
        case 'b':
        printf("B형입니다.\n");
        break;
        
        case 'O':
        case 'o':
        printf("O형입니다.\n");
        break;
        
        default:
        printf("입력오류\n");
        break;
    }

    return 0;
}
