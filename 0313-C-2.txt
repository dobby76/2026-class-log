#include <stdio.h>

int main()
{
    int num1,num2,sum,sub,mul;
    char op;
    double divi;
    
    printf("첫번째 숫자 입력: ");
    scanf("%d", &num1);
    
    printf("연산자 입력(+ - * /): ");
    scanf(" %c", &op);
    
    printf("두번째 숫자 입력: ");
    scanf("%d", &num2);
    
    switch(op)
    {
        case '+':
        sum = num1 + num2;
        printf("%d\n", sum);
        break;
        
        case '-':
        sub = num1 - num2;
        printf("%d\n", sub);
        break;
        
        case '*':
        mul = num1 * num2;
        printf("%d\n", mul);
        break;
        
        case '/':
        divi = (double)num1 / num2;
        printf("%f\n", divi);
        break;
        
        default:
        printf("입력오류");
    }

    return 0;
}
